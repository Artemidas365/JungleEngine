41template <class T>
42int testForAxisAngle(glm::vec<3, T, glm::defaultp> const axisTrue, T const angleTrue)
43{
44 T const eps = std::sqrt(std::numeric_limits<T>::epsilon());
45
46 glm::mat<4, 4, T, glm::defaultp> const matTrue = glm::axisAngleMatrix(axisTrue, angleTrue);
47
48 glm::vec<3, T, glm::defaultp> axis;
49 T angle;
50 glm::axisAngle(matTrue, axis, angle);
51 glm::mat<4, 4, T, glm::defaultp> const matRebuilt = glm::axisAngleMatrix(axis, angle);
52
53 glm::mat<4, 4, T, glm::defaultp> const errMat = matTrue - matRebuilt;
54 T const maxErr = glm::compMax(glm::vec<4, T, glm::defaultp>(
55 glm::compMax(glm::abs(errMat[0])),
56 glm::compMax(glm::abs(errMat[1])),
57 glm::compMax(glm::abs(errMat[2])),
58 glm::compMax(glm::abs(errMat[3]))
59 ));
60
61 return maxErr < eps ? 0 : 1;
62}
