8int pop(unsigned x) {
9 x = x - ((x >> 1) & 0x55555555);
10 x = (x & 0x33333333) + ((x >> 2) & 0x33333333);
11 x = (x + (x >> 4)) & 0x0F0F0F0F;
12 x = x + (x << 8);
13 x = x + (x << 16);
14 return x >> 24;
15}

17int nlz1(unsigned x) {
18 int n;
19
20 if (x == 0) return(32);
21 n = 0;
22 if (x <= 0x0000FFFF) {n = n +16; x = x <<16;}
23 if (x <= 0x00FFFFFF) {n = n + 8; x = x << 8;}
24 if (x <= 0x0FFFFFFF) {n = n + 4; x = x << 4;}
25 if (x <= 0x3FFFFFFF) {n = n + 2; x = x << 2;}
26 if (x <= 0x7FFFFFFF) {n = n + 1;}
27 return n;
28}

30int nlz1a(unsigned x) {
31 int n;
32
33/* if (x == 0) return(32); */
34 if (static_cast<int>(x) <= 0) return (~x >> 26) & 32;
35 n = 1;
36 if ((x >> 16) == 0) {n = n +16; x = x <<16;}
37 if ((x >> 24) == 0) {n = n + 8; x = x << 8;}
38 if ((x >> 28) == 0) {n = n + 4; x = x << 4;}
39 if ((x >> 30) == 0) {n = n + 2; x = x << 2;}
40 n = n - (x >> 31);
41 return n;
42}

45int nlz2(unsigned x) {
46 unsigned y;
47 int n;
48
49 n = 32;
50 y = x >>16; if (y != 0) {n = n -16; x = y;}
51 y = x >> 8; if (y != 0) {n = n - 8; x = y;}
52 y = x >> 4; if (y != 0) {n = n - 4; x = y;}
53 y = x >> 2; if (y != 0) {n = n - 2; x = y;}
54 y = x >> 1; if (y != 0) return n - 2;
55 return n - x;
56}

58// As above but coded as a loop for compactness:
59// 23 to 33 basic Risc instructions.
60int nlz2a(unsigned x) {
61 unsigned y;
62 int n, c;
63
64 n = 32;
65 c = 16;
66 do {
67 y = x >> c; if (y != 0) {n = n - c; x = y;}
68 c = c >> 1;
69 } while (c != 0);
70 return n - x;
71}

73int nlz3(int x) {
74 int y, n;
75
76 n = 0;
77 y = x;
78L: if (x < 0) return n;
79 if (y == 0) return 32 - n;
80 n = n + 1;
81 x = x << 1;
82 y = y >> 1;
83 goto L;

86int nlz4(unsigned x) {
87 int y, m, n;
88
89 y = -(x >> 16); // If left half of x is 0,
90 m = (y >> 16) & 16; // set n = 16. If left half
91 n = 16 - m; // is nonzero, set n = 0 and
92 x = x >> m; // shift x right 16.
93 // Now x is of the form 0000xxxx.
94 y = x - 0x100; // If positions 8-15 are 0,
95 m = (y >> 16) & 8; // add 8 to n and shift x left 8.
96 n = n + m;
97 x = x << m;
98
99 y = x - 0x1000; // If positions 12-15 are 0,
100 m = (y >> 16) & 4; // add 4 to n and shift x left 4.
101 n = n + m;
102 x = x << m;
103
104 y = x - 0x4000; // If positions 14-15 are 0,
105 m = (y >> 16) & 2; // add 2 to n and shift x left 2.
106 n = n + m;
107 x = x << m;
108
109 y = x >> 14; // Set y = 0, 1, 2, or 3.
110 m = y & ~(y >> 1); // Set m = 0, 1, 2, or 2 resp.
111 return n + 2 - m;
112}

125/* The four programs below are not valid ANSI C programs. This is
126because they refer to the same storage locations as two different types.
127However, they work with xlc/AIX, gcc/AIX, and gcc/NT. If you try to
128code them more compactly by declaring a variable xx to be "double," and
129then using
130
131 n = 1054 - (*((unsigned *)&xx + LE) >> 20);
132
133then you are violating not only the rule above, but also the ANSI C
134rule that pointer arithmetic can be performed only on pointers to
135array elements.
136 When coded with the above statement, the program fails with xlc,
137gcc/AIX, and gcc/NT, at some optimization levels.
138 BTW, these programs use the "anonymous union" feature of C++, not
139available in C. */
140
141int nlz6(unsigned k)
142{
143 union {
144 unsigned asInt[2];
145 double asDouble;
146 };
147 int n;
148
149 asDouble = static_cast<double>(k) + 0.5;
150 n = 1054 - (asInt[LE] >> 20);
151 return n;
152}

168 /* In single qualifier, round-to-nearest mode, the basic method fails for:
169 k = 0, k = 01FFFFFF, 03FFFFFE <= k <= 03FFFFFF,
170 07FFFFFC <= k <= 07FFFFFF,
171 0FFFFFF8 <= k <= 0FFFFFFF,
172 ...
173 7FFFFFC0 <= k <= 7FFFFFFF.
174 FFFFFF80 <= k <= FFFFFFFF.
175 For k = 0 it gives 158, and for the other values it is too low by 1. */
176
177int nlz8(unsigned k)
178{
179 union {
180 unsigned asInt;
181 float asFloat;
182 };
183 int n;
184
185 k = k & ~(k >> 1); /* Fix problem with rounding. */
186 asFloat = static_cast<float>(k) + 0.5f;
187 n = 158 - (asInt >> 23);
188 return n;
189}

191/* The example below shows how to make a macro for nlz. It uses an
192extension to the C and C++ languages that is provided by the GNU C/C++
193compiler, namely, that of allowing statements and declarations in
194expressions (see "Using and Porting GNU CC", by Richard M. Stallman
195(1998). The underscores are necessary to protect against the
196possibility that the macro argument will conflict with one of its local
197variables, e.g., NLZ(k). */

199int nlz9(unsigned k)
200{
201 union {
202 unsigned asInt;
203 float asFloat;
204 };
205 int n;

207 k = k & ~(k >> 1); /* Fix problem with rounding. */
208 asFloat = static_cast<float>(k);
209 n = 158 - (asInt >> 23);
210 n = (n & 31) + (n >> 6); /* Fix problem with k = 0. */
211 return n;

214/* Below are three nearly equivalent programs for computing the number
215of leading zeros in a word. This material is not in HD, but may be in a
216future edition.
217 Immediately below is Robert Harley's algorithm, found at the
218comp.arch newsgroup entry dated 7/12/96, pointed out to me by Norbert
219Juffa.
220 Table entries marked "u" are unused. 14 ops including a multiply,
221plus an indexed load.
222 The smallest multiplier that works is 0x045BCED1 = 17*65*129*513 (all
223of form 2**k + 1). There are no multipliers of three terms of the form
2242**k +- 1 that work, with a table size of 64 or 128. There are some,
225with a table size of 64, if you precede the multiplication with x = x -
226(x >> 1), but that seems less elegant. There are also some if you use a
227table size of 256, the smallest is 0x01033CBF = 65*255*1025 (this would
228save two instructions in the form of this algorithm with the
229multiplication expanded into shifts and adds, but the table size is
230getting a bit large). */
231
232#define u 99
233int nlz10(unsigned x)
234{
235 static char table[64] =
236 {32,31, u,16, u,30, 3, u, 15, u, u, u,29,10, 2, u,
237 u, u,12,14,21, u,19, u, u,28, u,25, u, 9, 1, u,
238 17, u, 4, u, u, u,11, u, 13,22,20, u,26, u, u,18,
239 5, u, u,23, u,27, u, 6, u,24, 7, u, 8, u, 0, u};
240
241 x = x | (x >> 1); // Propagate leftmost
242 x = x | (x >> 2); // 1-bit to the right.
243 x = x | (x >> 4);
244 x = x | (x >> 8);
245 x = x | (x >>16);
246 x = x*0x06EB14F9; // Multiplier is 7*255**3.
247 return table[x >> 26];
248}

250/* Harley's algorithm with multiply expanded.
25119 elementary ops plus an indexed load. */
252
253int nlz10a(unsigned x)
254{
255 static char table[64] =
256 {32,31, u,16, u,30, 3, u, 15, u, u, u,29,10, 2, u,
257 u, u,12,14,21, u,19, u, u,28, u,25, u, 9, 1, u,
258 17, u, 4, u, u, u,11, u, 13,22,20, u,26, u, u,18,
259 5, u, u,23, u,27, u, 6, u,24, 7, u, 8, u, 0, u};
260
261 x = x | (x >> 1); // Propagate leftmost
262 x = x | (x >> 2); // 1-bit to the right.
263 x = x | (x >> 4);
264 x = x | (x >> 8);
265 x = x | (x >> 16);
266 x = (x << 3) - x; // Multiply by 7.
267 x = (x << 8) - x; // Multiply by 255.
268 x = (x << 8) - x; // Again.
269 x = (x << 8) - x; // Again.
270 return table[x >> 26];
271}

273/* Julius Goryavsky's version of Harley's algorithm.
27417 elementary ops plus an indexed load, if the machine
275has "and not." */
276
277int nlz10b(unsigned x)
278{
279 static char table[64] =
280 {32,20,19, u, u,18, u, 7, 10,17, u, u,14, u, 6, u,
281 u, 9, u,16, u, u, 1,26, u,13, u, u,24, 5, u, u,
282 u,21, u, 8,11, u,15, u, u, u, u, 2,27, 0,25, u,
283 22, u,12, u, u, 3,28, u, 23, u, 4,29, u, u,30,31};
284
285 x = x | (x >> 1); // Propagate leftmost
286 x = x | (x >> 2); // 1-bit to the right.
287 x = x | (x >> 4);
288 x = x | (x >> 8);
289 x = x & ~(x >> 16);
290 x = x*0xFD7049FF; // Activate this line or the following 3.
291 // x = (x << 9) - x; // Multiply by 511.
292 // x = (x << 11) - x; // Multiply by 2047.
293 // x = (x << 14) - x; // Multiply by 16383.
294 return table[x >> 26];
295}
