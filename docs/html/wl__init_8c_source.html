97static void registryHandleGlobal(void* userData,
98 struct wl_registry* registry,
99 uint32_t name,
100 const char* interface,
101 uint32_t version)
102{
103 if (strcmp(interface, "wl_compositor") == 0)
104 {
105 _glfw.wl.compositor =
106 wl_registry_bind(registry, name, &wl_compositor_interface,
107 _glfw_min(3, version));
108 }
109 else if (strcmp(interface, "wl_subcompositor") == 0)
110 {
111 _glfw.wl.subcompositor =
112 wl_registry_bind(registry, name, &wl_subcompositor_interface, 1);
113 }
114 else if (strcmp(interface, "wl_shm") == 0)
115 {
116 _glfw.wl.shm =
117 wl_registry_bind(registry, name, &wl_shm_interface, 1);
118 }
119 else if (strcmp(interface, "wl_output") == 0)
120 {
121 _glfwAddOutputWayland(name, version);
122 }
123 else if (strcmp(interface, "wl_seat") == 0)
124 {
125 if (!_glfw.wl.seat)
126 {
127 _glfw.wl.seat =
128 wl_registry_bind(registry, name, &wl_seat_interface,
129 _glfw_min(4, version));
130 _glfwAddSeatListenerWayland(_glfw.wl.seat);
131 }
132 }
133 else if (strcmp(interface, "wl_data_device_manager") == 0)
134 {
135 if (!_glfw.wl.dataDeviceManager)
136 {
137 _glfw.wl.dataDeviceManager =
138 wl_registry_bind(registry, name,
139 &wl_data_device_manager_interface, 1);
140 }
141 }
142 else if (strcmp(interface, "xdg_wm_base") == 0)
143 {
144 _glfw.wl.wmBase =
145 wl_registry_bind(registry, name, &xdg_wm_base_interface, 1);
146 xdg_wm_base_add_listener(_glfw.wl.wmBase, &wmBaseListener, NULL);
147 }
148 else if (strcmp(interface, "zxdg_decoration_manager_v1") == 0)
149 {
150 _glfw.wl.decorationManager =
151 wl_registry_bind(registry, name,
152 &zxdg_decoration_manager_v1_interface,
153 1);
154 }
155 else if (strcmp(interface, "wp_viewporter") == 0)
156 {
157 _glfw.wl.viewporter =
158 wl_registry_bind(registry, name, &wp_viewporter_interface, 1);
159 }
160 else if (strcmp(interface, "zwp_relative_pointer_manager_v1") == 0)
161 {
162 _glfw.wl.relativePointerManager =
163 wl_registry_bind(registry, name,
164 &zwp_relative_pointer_manager_v1_interface,
165 1);
166 }
167 else if (strcmp(interface, "zwp_pointer_constraints_v1") == 0)
168 {
169 _glfw.wl.pointerConstraints =
170 wl_registry_bind(registry, name,
171 &zwp_pointer_constraints_v1_interface,
172 1);
173 }
174 else if (strcmp(interface, "zwp_idle_inhibit_manager_v1") == 0)
175 {
176 _glfw.wl.idleInhibitManager =
177 wl_registry_bind(registry, name,
178 &zwp_idle_inhibit_manager_v1_interface,
179 1);
180 }
181}

183static void registryHandleGlobalRemove(void* userData,
184 struct wl_registry* registry,
185 uint32_t name)
186{
187 for (int i = 0; i < _glfw.monitorCount; ++i)

352static GLFWbool loadCursorTheme(void)
353{
354 int cursorSize = 16;
355
356 const char* sizeString = getenv("XCURSOR_SIZE");
357 if (sizeString)
358 {
359 errno = 0;
360 const long cursorSizeLong = strtol(sizeString, NULL, 10);
361 if (errno == 0 && cursorSizeLong > 0 && cursorSizeLong < INT_MAX)
362 cursorSize = (int) cursorSizeLong;
363 }
364
365 const char* themeName = getenv("XCURSOR_THEME");
366
367 _glfw.wl.cursorTheme = wl_cursor_theme_load(themeName, cursorSize, _glfw.wl.shm);
368 if (!_glfw.wl.cursorTheme)
369 {
370 _glfwInputError(GLFW_PLATFORM_ERROR,
371 "Wayland: Failed to load default cursor theme");
372 return GLFW_FALSE;
373 }
374
375 // If this happens to be NULL, we just fallback to the scale=1 version.
376 _glfw.wl.cursorThemeHiDPI =
377 wl_cursor_theme_load(themeName, cursorSize * 2, _glfw.wl.shm);
378
379 _glfw.wl.cursorSurface = wl_compositor_create_surface(_glfw.wl.compositor);
380 _glfw.wl.cursorTimerfd = timerfd_create(CLOCK_MONOTONIC, TFD_CLOEXEC | TFD_NONBLOCK);
381 return GLFW_TRUE;
382}
