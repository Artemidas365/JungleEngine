10 void Camera::updateCameraVectors() {
11 // calculate the new Front vector
12 glm::vec3 front;
13 front.x = cos(glm::radians(yaw)) * cos(glm::radians(pitch));
14 front.y = sin(glm::radians(pitch));
15 front.z = sin(glm::radians(yaw)) * cos(glm::radians(pitch));
16 camFront = glm::normalize(front);
17 // also re-calculate the Right and Up vector
18 camRight = glm::normalize(glm::cross(camFront, worldUp)); // normalize the vectors, because their length gets closer to 0 the more you look up or down which results in slower movement.
19 camUp = glm::normalize(glm::cross(camRight, camFront));
20 }

51 void Camera::ProcessKeyboard(Camera_Movement direction, float deltaTime)
52 {
53 float velocity = movementSpeed * deltaTime;
54 if (direction == FORWARD)
55 camPos += camFront * velocity;
56 if (direction == BACKWARD)
57 camPos -= camFront * velocity;
58 if (direction == LEFT)
59 camPos -= camRight * velocity;
60 if (direction == RIGHT)
61 camPos += camRight * velocity;
62 }

64 void Camera::ProcessMouseMovement(float Xoffset, float Yoffset, GLboolean constrainPitch)
65 {
66 Xoffset *= mouseSensitivity;
67 Yoffset *= mouseSensitivity;
68
69 yaw = glm::mod( yaw + Xoffset, 360.0f );
70 pitch += Yoffset;
71
72 // make sure that when pitch is out of bounds, screen doesn't get flipped
73 if (constrainPitch)
74 {
75 if (pitch > 89.0f)
76 pitch = 89.0f;
77 if (pitch < -89.0f)
78 pitch = -89.0f;
79 }
80
81 // update Front, Right and Up Vectors using the updated Euler angles
82 updateCameraVectors();
83 }
