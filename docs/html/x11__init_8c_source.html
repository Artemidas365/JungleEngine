362 // Find the X11 key code -> GLFW key code mapping
363 for (int scancode = scancodeMin; scancode <= scancodeMax; scancode++)
364 {
365 int key = GLFW_KEY_UNKNOWN;
366
367 // Map the key name to a GLFW key code. Note: We use the US
368 // keyboard layout. Because function keys aren't mapped correctly
369 // when using traditional KeySym translations, they are mapped
370 // here instead.
371 for (int i = 0; i < sizeof(keymap) / sizeof(keymap[0]); i++)
372 {
373 if (strncmp(desc->names->keys[scancode].name,
374 keymap[i].name,
375 XkbKeyNameLength) == 0)
376 {
377 key = keymap[i].key;
378 break;
379 }
380 }
381
382 // Fall back to key aliases in case the key name did not match
383 for (int i = 0; i < desc->names->num_key_aliases; i++)
384 {
385 if (key != GLFW_KEY_UNKNOWN)
386 break;
387
388 if (strncmp(desc->names->key_aliases[i].real,
389 desc->names->keys[scancode].name,
390 XkbKeyNameLength) != 0)
391 {
392 continue;
393 }
394
395 for (int j = 0; j < sizeof(keymap) / sizeof(keymap[0]); j++)
396 {
397 if (strncmp(desc->names->key_aliases[i].alias,
398 keymap[j].name,
399 XkbKeyNameLength) == 0)
400 {
401 key = keymap[j].key;
402 break;
403 }
404 }
405 }
406
407 _glfw.x11.keycodes[scancode] = key;
408 }
409
410 XkbFreeNames(desc, XkbKeyNamesMask, True);
411 XkbFreeKeyboard(desc, 0, True);
412 }
413 else
414 XDisplayKeycodes(_glfw.x11.display, &scancodeMin, &scancodeMax);
415
416 int width;
417 KeySym* keysyms = XGetKeyboardMapping(_glfw.x11.display,
418 scancodeMin,
419 scancodeMax - scancodeMin + 1,
420 &width);
421
422 for (int scancode = scancodeMin; scancode <= scancodeMax; scancode++)
423 {
424 // Translate the un-translated key codes using traditional X11 KeySym
425 // lookups
426 if (_glfw.x11.keycodes[scancode] < 0)
427 {
428 const size_t base = (scancode - scancodeMin) * width;
429 _glfw.x11.keycodes[scancode] = translateKeySyms(&keysyms[base], width);
430 }
431
432 // Store the reverse translation for faster key name lookup
433 if (_glfw.x11.keycodes[scancode] > 0)
434 _glfw.x11.scancodes[_glfw.x11.keycodes[scancode]] = scancode;
435 }
436
437 XFree(keysyms);
438}

440// Check whether the IM has a usable style
441//
442static GLFWbool hasUsableInputMethodStyle(void)
443{
444 GLFWbool found = GLFW_FALSE;
445 XIMStyles* styles = NULL;
446
447 if (XGetIMValues(_glfw.x11.im, XNQueryInputStyle, &styles, NULL) != NULL)
448 return GLFW_FALSE;
449
450 for (unsigned int i = 0; i < styles->count_styles; i++)
451 {
452 if (styles->supported_styles[i] == (XIMPreeditNothing | XIMStatusNothing))
453 {
454 found = GLFW_TRUE;
455 break;
456 }
457 }
458
459 XFree(styles);
460 return found;
461}

492 for (_GLFWwindow* window = _glfw.windowListHead; window; window = window->next)
493 _glfwCreateInputContextX11(window);
494 }

509 }
510
511 return None;

514// Check whether the running window manager is EWMH-compliant
515//
516static void detectEWMH(void)
517{
518 // First we read the _NET_SUPPORTING_WM_CHECK property on the root window
519
520 Window* windowFromRoot = NULL;
521 if (!_glfwGetWindowPropertyX11(_glfw.x11.root,
522 _glfw.x11.NET_SUPPORTING_WM_CHECK,
523 XA_WINDOW,
524 (unsigned char**) &windowFromRoot))
525 {
526 return;
527 }
528
529 _glfwGrabErrorHandlerX11();
530
531 // If it exists, it should be the XID of a top-level window
532 // Then we look for the same property on that window
533
534 Window* windowFromChild = NULL;
535 if (!_glfwGetWindowPropertyX11(*windowFromRoot,
536 _glfw.x11.NET_SUPPORTING_WM_CHECK,
537 XA_WINDOW,
538 (unsigned char**) &windowFromChild))
539 {
540 XFree(windowFromRoot);
541 return;
542 }
543
544 _glfwReleaseErrorHandlerX11();
545
546 // If the property exists, it should contain the XID of the window
547
548 if (*windowFromRoot != *windowFromChild)
549 {
550 XFree(windowFromRoot);
551 XFree(windowFromChild);
552 return;
553 }
554
555 XFree(windowFromRoot);
556 XFree(windowFromChild);
557
558 // We are now fairly sure that an EWMH-compliant WM is currently running
559 // We can now start querying the WM about what features it supports by
560 // looking in the _NET_SUPPORTED property on the root window
561 // It should contain a list of supported EWMH protocol and state atoms
562
563 Atom* supportedAtoms = NULL;
564 const unsigned long atomCount =
565 _glfwGetWindowPropertyX11(_glfw.x11.root,
566 _glfw.x11.NET_SUPPORTED,
567 XA_ATOM,
568 (unsigned char**) &supportedAtoms);
569
570 // See which of the atoms we support that are supported by the WM
571
572 _glfw.x11.NET_WM_STATE =
573 getAtomIfSupported(supportedAtoms, atomCount, "_NET_WM_STATE");
574 _glfw.x11.NET_WM_STATE_ABOVE =
575 getAtomIfSupported(supportedAtoms, atomCount, "_NET_WM_STATE_ABOVE");
576 _glfw.x11.NET_WM_STATE_FULLSCREEN =
577 getAtomIfSupported(supportedAtoms, atomCount, "_NET_WM_STATE_FULLSCREEN");
578 _glfw.x11.NET_WM_STATE_MAXIMIZED_VERT =
579 getAtomIfSupported(supportedAtoms, atomCount, "_NET_WM_STATE_MAXIMIZED_VERT");

992// Retrieve system content scale via folklore heuristics
993//
994static void getSystemContentScale(float* xscale, float* yscale)
995{
996 // Start by assuming the default X11 DPI
997 // NOTE: Some desktop environments (KDE) may remove the Xft.dpi field when it
998 // would be set to 96, so assume that is the case if we cannot find it
999 float xdpi = 96.f, ydpi = 96.f;
1000
1001 // NOTE: Basing the scale on Xft.dpi where available should provide the most
1002 // consistent user experience (matches Qt, Gtk, etc), although not
1003 // always the most accurate one
1004 char* rms = XResourceManagerString(_glfw.x11.display);
1005 if (rms)
1006 {
1007 XrmDatabase db = XrmGetStringDatabase(rms);
1008 if (db)
1009 {
1010 XrmValue value;
1011 char* type = NULL;
1012
1013 if (XrmGetResource(db, "Xft.dpi", "Xft.Dpi", &type, &value))
1014 {
1015 if (type && strcmp(type, "String") == 0)
1016 xdpi = ydpi = atof(value.addr);
1017 }
1018
1019 XrmDestroyDatabase(db);
1020 }
1021 }
1022
1023 *xscale = xdpi / 96.f;
1024 *yscale = ydpi / 96.f;
1025}

1058 strerror(errno));
1059 return GLFW_FALSE;
1060 }
1061
1062 for (int i = 0; i < 2; i++)
1063 {
1064 const int sf = fcntl(_glfw.x11.emptyEventPipe[i], F_GETFL, 0);
1065 const int df = fcntl(_glfw.x11.emptyEventPipe[i], F_GETFD, 0);
1066
1067 if (sf == -1 || df == -1 ||
1068 fcntl(_glfw.x11.emptyEventPipe[i], F_SETFL, sf | O_NONBLOCK) == -1 ||
1069 fcntl(_glfw.x11.emptyEventPipe[i], F_SETFD, df | FD_CLOEXEC) == -1)
1070 {
1071 _glfwInputError(GLFW_PLATFORM_ERROR,
1072 "X11: Failed to set flags for empty event pipe: %s",
1073 strerror(errno));
1074 return GLFW_FALSE;
1075 }
1076 }
1077
1078 return GLFW_TRUE;
1079}

1128//
1129Cursor _glfwCreateNativeCursorX11(const GLFWimage* image, int xhot, int yhot)
1130{
1131 Cursor cursor;
1132
1133 if (!_glfw.x11.xcursor.handle)
1134 return None;
1135
1136 XcursorImage* native = XcursorImageCreate(image->width, image->height);
1137 if (native == NULL)
1138 return None;
1139
1140 native->xhot = xhot;
1141 native->yhot = yhot;
1142
1143 unsigned char* source = (unsigned char*) image->pixels;
1144 XcursorPixel* target = native->pixels;
1145
1146 for (int i = 0; i < image->width * image->height; i++, target++, source += 4)
1147 {
1148 unsigned int alpha = source[3];
1149
1150 *target = (alpha << 24) |
1151 ((unsigned char) ((source[0] * alpha) / 255) << 16) |
1152 ((unsigned char) ((source[1] * alpha) / 255) << 8) |
1153 ((unsigned char) ((source[2] * alpha) / 255) << 0);
1154 }
1155
1156 cursor = XcursorImageLoadCursor(_glfw.x11.display, native);
1157 XcursorImageDestroy(native);
1158
1159 return cursor;
1160}
