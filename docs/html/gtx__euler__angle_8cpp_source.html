145 bool epsilonEqual(glm::mat4 const& mat1, glm::mat4 const& mat2, glm::mat4::value_type const& epsilon)
146 {
147 return glm::all(glm::epsilonEqual(mat1[0], mat2[0], epsilon)) ?
148 (
149 glm::all(glm::epsilonEqual(mat1[1], mat2[1], epsilon)) ?
150 (
151 glm::all(glm::epsilonEqual(mat1[2], mat2[2], epsilon)) ?
152 (
153 glm::all(glm::epsilonEqual(mat1[3], mat2[3], epsilon)) ? true : false
154 ) : false
155 ) : false
156 ) : false;

159 template<typename RotationFunc, typename TestDerivedFunc>
160 int test(RotationFunc rotationFunc, TestDerivedFunc testDerivedFunc, const glm::vec3& basis)
161 {
162 int Error = 0;
163
164 typedef glm::vec3::value_type value;
165 value const zeroAngle(0.0f);
166 value const Angle(glm::pi<float>() * 0.75f);
167 value const negativeAngle(-Angle);
168 value const zeroAngleVelocity(0.0f);
169 value const AngleVelocity(glm::pi<float>() * 0.27f);
170 value const negativeAngleVelocity(-AngleVelocity);
171
172 typedef std::pair<value,value> AngleAndAngleVelocity;
173 std::vector<AngleAndAngleVelocity> testPairs;
174 testPairs.push_back(AngleAndAngleVelocity(zeroAngle, zeroAngleVelocity));
175 testPairs.push_back(AngleAndAngleVelocity(zeroAngle, AngleVelocity));
176 testPairs.push_back(AngleAndAngleVelocity(zeroAngle, negativeAngleVelocity));
177 testPairs.push_back(AngleAndAngleVelocity(Angle, zeroAngleVelocity));
178 testPairs.push_back(AngleAndAngleVelocity(Angle, AngleVelocity));
179 testPairs.push_back(AngleAndAngleVelocity(Angle, negativeAngleVelocity));
180 testPairs.push_back(AngleAndAngleVelocity(negativeAngle, zeroAngleVelocity));
181 testPairs.push_back(AngleAndAngleVelocity(negativeAngle, AngleVelocity));
182 testPairs.push_back(AngleAndAngleVelocity(negativeAngle, negativeAngleVelocity));
183
184 for (size_t i = 0, size = testPairs.size(); i < size; ++i)
185 {
186 AngleAndAngleVelocity const& pair = testPairs.at(i);
187
188 glm::mat4 const W = glm::matrixCross4(basis * pair.second);
189 glm::mat4 const rotMt = glm::transpose(rotationFunc(pair.first));
190 glm::mat4 const derivedRotM = testDerivedFunc(pair.first, pair.second);
191
192 Error += epsilonEqual(W, derivedRotM * rotMt, 0.00001f) ? 0 : 1;
193 }
194
195 return Error;
196 }
