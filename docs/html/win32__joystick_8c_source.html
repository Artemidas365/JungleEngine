149// Returns a description fitting the specified XInput capabilities
150//
151static const char* getDeviceDescription(const XINPUT_CAPABILITIES* xic)
152{
153 switch (xic->SubType)
154 {
155 case XINPUT_DEVSUBTYPE_WHEEL:
156 return "XInput Wheel";
157 case XINPUT_DEVSUBTYPE_ARCADE_STICK:
158 return "XInput Arcade Stick";
159 case XINPUT_DEVSUBTYPE_FLIGHT_STICK:
160 return "XInput Flight Stick";
161 case XINPUT_DEVSUBTYPE_DANCE_PAD:
162 return "XInput Dance Pad";
163 case XINPUT_DEVSUBTYPE_GUITAR:
164 return "XInput Guitar";
165 case XINPUT_DEVSUBTYPE_DRUM_KIT:
166 return "XInput Drum Kit";
167 case XINPUT_DEVSUBTYPE_GAMEPAD:
168 {
169 if (xic->Flags & XINPUT_CAPS_WIRELESS)
170 return "Wireless Xbox Controller";
171 else
172 return "Xbox Controller";
173 }
174 }
175
176 return "Unknown XInput Device";
177}

179// Lexically compare device objects
180//
181static int compareJoystickObjects(const void* first, const void* second)
182{
183 const _GLFWjoyobjectWin32* fo = first;
184 const _GLFWjoyobjectWin32* so = second;
185
186 if (fo->type != so->type)
187 return fo->type - so->type;
188
189 return fo->offset - so->offset;
190}

192// Checks whether the specified device supports XInput
193// Technique from FDInputJoystickManager::IsXInputDeviceFast in ZDoom
194//
195static GLFWbool supportsXInput(const GUID* guid)
196{
197 UINT i, count = 0;
198 RAWINPUTDEVICELIST* ridl;
199 GLFWbool result = GLFW_FALSE;
200
201 if (GetRawInputDeviceList(NULL, &count, sizeof(RAWINPUTDEVICELIST)) != 0)
202 return GLFW_FALSE;
203
204 ridl = _glfw_calloc(count, sizeof(RAWINPUTDEVICELIST));
205
206 if (GetRawInputDeviceList(ridl, &count, sizeof(RAWINPUTDEVICELIST)) == (UINT) -1)
207 {
208 _glfw_free(ridl);
209 return GLFW_FALSE;
210 }
211
212 for (i = 0; i < count; i++)
213 {
214 RID_DEVICE_INFO rdi;
215 char name[256];
216 UINT size;
217
218 if (ridl[i].dwType != RIM_TYPEHID)
219 continue;
220
221 ZeroMemory(&rdi, sizeof(rdi));
222 rdi.cbSize = sizeof(rdi);
223 size = sizeof(rdi);
224
225 if ((INT) GetRawInputDeviceInfoA(ridl[i].hDevice,
226 RIDI_DEVICEINFO,
227 &rdi, &size) == -1)
228 {
229 continue;
230 }
231
232 if (MAKELONG(rdi.hid.dwVendorId, rdi.hid.dwProductId) != (LONG) guid->Data1)
233 continue;
234
235 memset(name, 0, sizeof(name));
236 size = sizeof(name);
237
238 if ((INT) GetRawInputDeviceInfoA(ridl[i].hDevice,
239 RIDI_DEVICENAME,
240 name, &size) == -1)
241 {
242 break;
243 }
244
245 name[sizeof(name) - 1] = '\0';
246 if (strstr(name, "IG_"))
247 {
248 result = GLFW_TRUE;
249 break;
250 }
251 }
252
253 _glfw_free(ridl);
254 return result;
255}

257// Frees all resources associated with the specified joystick
258//
259static void closeJoystick(_GLFWjoystick* js)
260{
261 _glfwInputJoystick(js, GLFW_DISCONNECTED);
262
263 if (js->win32.device)
264 {
265 IDirectInputDevice8_Unacquire(js->win32.device);
266 IDirectInputDevice8_Release(js->win32.device);
267 }
268
269 _glfw_free(js->win32.objects);
270 _glfwFreeJoystick(js);
271}

273// DirectInput device object enumeration callback
274// Insights gleaned from SDL
275//
276static BOOL CALLBACK deviceObjectCallback(const DIDEVICEOBJECTINSTANCEW* doi,
277 void* user)
278{
279 _GLFWobjenumWin32* data = user;
280 _GLFWjoyobjectWin32* object = data->objects + data->objectCount;
281
282 if (DIDFT_GETTYPE(doi->dwType) & DIDFT_AXIS)
283 {
284 DIPROPRANGE dipr;
285
286 if (memcmp(&doi->guidType, &GUID_Slider, sizeof(GUID)) == 0)
287 object->offset = DIJOFS_SLIDER(data->sliderCount);
288 else if (memcmp(&doi->guidType, &GUID_XAxis, sizeof(GUID)) == 0)
289 object->offset = DIJOFS_X;
290 else if (memcmp(&doi->guidType, &GUID_YAxis, sizeof(GUID)) == 0)
291 object->offset = DIJOFS_Y;
292 else if (memcmp(&doi->guidType, &GUID_ZAxis, sizeof(GUID)) == 0)
293 object->offset = DIJOFS_Z;
294 else if (memcmp(&doi->guidType, &GUID_RxAxis, sizeof(GUID)) == 0)
295 object->offset = DIJOFS_RX;
296 else if (memcmp(&doi->guidType, &GUID_RyAxis, sizeof(GUID)) == 0)
297 object->offset = DIJOFS_RY;
298 else if (memcmp(&doi->guidType, &GUID_RzAxis, sizeof(GUID)) == 0)
299 object->offset = DIJOFS_RZ;
300 else
301 return DIENUM_CONTINUE;
302
303 ZeroMemory(&dipr, sizeof(dipr));
304 dipr.diph.dwSize = sizeof(dipr);
305 dipr.diph.dwHeaderSize = sizeof(dipr.diph);
306 dipr.diph.dwObj = doi->dwType;
307 dipr.diph.dwHow = DIPH_BYID;
308 dipr.lMin = -32768;
309 dipr.lMax = 32767;
310
311 if (FAILED(IDirectInputDevice8_SetProperty(data->device,
312 DIPROP_RANGE,
313 &dipr.diph)))
314 {
315 return DIENUM_CONTINUE;
316 }
317
318 if (memcmp(&doi->guidType, &GUID_Slider, sizeof(GUID)) == 0)
319 {
320 object->type = _GLFW_TYPE_SLIDER;
321 data->sliderCount++;
322 }
323 else
324 {
325 object->type = _GLFW_TYPE_AXIS;
326 data->axisCount++;
327 }
328 }
329 else if (DIDFT_GETTYPE(doi->dwType) & DIDFT_BUTTON)
330 {
331 object->offset = DIJOFS_BUTTON(data->buttonCount);
332 object->type = _GLFW_TYPE_BUTTON;
333 data->buttonCount++;
334 }
335 else if (DIDFT_GETTYPE(doi->dwType) & DIDFT_POV)
336 {
337 object->offset = DIJOFS_POV(data->povCount);
338 object->type = _GLFW_TYPE_POV;
339 data->povCount++;
340 }
341
342 data->objectCount++;
343 return DIENUM_CONTINUE;
344}

346// DirectInput device enumeration callback
347//
348static BOOL CALLBACK deviceCallback(const DIDEVICEINSTANCE* di, void* user)
349{
350 int jid = 0;
351 DIDEVCAPS dc;
352 DIPROPDWORD dipd;
353 IDirectInputDevice8* device;
354 _GLFWobjenumWin32 data;
355 _GLFWjoystick* js;
356 char guid[33];
357 char name[256];
358
359 for (jid = 0; jid <= GLFW_JOYSTICK_LAST; jid++)
360 {
361 js = _glfw.joysticks + jid;
362 if (js->connected)
363 {
364 if (memcmp(&js->win32.guid, &di->guidInstance, sizeof(GUID)) == 0)
365 return DIENUM_CONTINUE;
366 }
367 }
368
369 if (supportsXInput(&di->guidProduct))
370 return DIENUM_CONTINUE;
371
372 if (FAILED(IDirectInput8_CreateDevice(_glfw.win32.dinput8.api,
373 &di->guidInstance,
374 &device,
375 NULL)))
376 {
377 _glfwInputError(GLFW_PLATFORM_ERROR, "Win32: Failed to create device");
378 return DIENUM_CONTINUE;
379 }
380
381 if (FAILED(IDirectInputDevice8_SetDataFormat(device, &_glfwDataFormat)))
382 {
383 _glfwInputError(GLFW_PLATFORM_ERROR,
384 "Win32: Failed to set device data format");
385
386 IDirectInputDevice8_Release(device);
387 return DIENUM_CONTINUE;
388 }
389
390 ZeroMemory(&dc, sizeof(dc));
391 dc.dwSize = sizeof(dc);
392
393 if (FAILED(IDirectInputDevice8_GetCapabilities(device, &dc)))
394 {
395 _glfwInputError(GLFW_PLATFORM_ERROR,
396 "Win32: Failed to query device capabilities");
397
398 IDirectInputDevice8_Release(device);
399 return DIENUM_CONTINUE;
400 }
401
402 ZeroMemory(&dipd, sizeof(dipd));
403 dipd.diph.dwSize = sizeof(dipd);
404 dipd.diph.dwHeaderSize = sizeof(dipd.diph);
405 dipd.diph.dwHow = DIPH_DEVICE;
406 dipd.dwData = DIPROPAXISMODE_ABS;
407
408 if (FAILED(IDirectInputDevice8_SetProperty(device,
409 DIPROP_AXISMODE,
410 &dipd.diph)))
411 {
412 _glfwInputError(GLFW_PLATFORM_ERROR,
413 "Win32: Failed to set device axis mode");
414
415 IDirectInputDevice8_Release(device);
416 return DIENUM_CONTINUE;
417 }
418
419 memset(&data, 0, sizeof(data));
420 data.device = device;
421 data.objects = _glfw_calloc(dc.dwAxes + (size_t) dc.dwButtons + dc.dwPOVs,
422 sizeof(_GLFWjoyobjectWin32));
423
424 if (FAILED(IDirectInputDevice8_EnumObjects(device,
425 deviceObjectCallback,
426 &data,
427 DIDFT_AXIS | DIDFT_BUTTON | DIDFT_POV)))
428 {
429 _glfwInputError(GLFW_PLATFORM_ERROR,
430 "Win32: Failed to enumerate device objects");
431
432 IDirectInputDevice8_Release(device);
433 _glfw_free(data.objects);
434 return DIENUM_CONTINUE;
435 }
436
437 qsort(data.objects, data.objectCount,
438 sizeof(_GLFWjoyobjectWin32),
439 compareJoystickObjects);
440
441 if (!WideCharToMultiByte(CP_UTF8, 0,
442 di->tszInstanceName, -1,
443 name, sizeof(name),
444 NULL, NULL))
445 {
446 _glfwInputError(GLFW_PLATFORM_ERROR,
447 "Win32: Failed to convert joystick name to UTF-8");
448
449 IDirectInputDevice8_Release(device);
450 _glfw_free(data.objects);
451 return DIENUM_STOP;
452 }
453
454 // Generate a joystick GUID that matches the SDL 2.0.5+ one
455 if (memcmp(&di->guidProduct.Data4[2], "PIDVID", 6) == 0)
456 {
457 sprintf(guid, "03000000%02x%02x0000%02x%02x000000000000",
458 (uint8_t) di->guidProduct.Data1,
459 (uint8_t) (di->guidProduct.Data1 >> 8),
460 (uint8_t) (di->guidProduct.Data1 >> 16),
461 (uint8_t) (di->guidProduct.Data1 >> 24));
462 }
463 else
464 {
465 sprintf(guid, "05000000%02x%02x%02x%02x%02x%02x%02x%02x%02x%02x%02x00",
466 name[0], name[1], name[2], name[3],
467 name[4], name[5], name[6], name[7],
468 name[8], name[9], name[10]);
469 }
470
471 js = _glfwAllocJoystick(name, guid,
472 data.axisCount + data.sliderCount,
473 data.buttonCount,
474 data.povCount);
475 if (!js)
476 {
477 IDirectInputDevice8_Release(device);
478 _glfw_free(data.objects);
479 return DIENUM_STOP;
480 }
481
482 js->win32.device = device;
483 js->win32.guid = di->guidInstance;
484 js->win32.objects = data.objects;
485 js->win32.objectCount = data.objectCount;
486
487 _glfwInputJoystick(js, GLFW_CONNECTED);
488 return DIENUM_CONTINUE;
489}

496// Checks for new joysticks after DBT_DEVICEARRIVAL
497//
498void _glfwDetectJoystickConnectionWin32(void)
499{
500 if (_glfw.win32.xinput.instance)
501 {
502 DWORD index;
503
504 for (index = 0; index < XUSER_MAX_COUNT; index++)
505 {
506 int jid;
507 char guid[33];
508 XINPUT_CAPABILITIES xic;
509 _GLFWjoystick* js;
510
511 for (jid = 0; jid <= GLFW_JOYSTICK_LAST; jid++)
512 {
513 if (_glfw.joysticks[jid].connected &&
514 _glfw.joysticks[jid].win32.device == NULL &&
515 _glfw.joysticks[jid].win32.index == index)
516 {
517 break;
518 }
519 }
520
521 if (jid <= GLFW_JOYSTICK_LAST)
522 continue;
523
524 if (XInputGetCapabilities(index, 0, &xic) != ERROR_SUCCESS)
525 continue;
526
527 // Generate a joystick GUID that matches the SDL 2.0.5+ one
528 sprintf(guid, "78696e707574%02x000000000000000000",
529 xic.SubType & 0xff);
530
531 js = _glfwAllocJoystick(getDeviceDescription(&xic), guid, 6, 10, 1);
532 if (!js)
533 continue;
534
535 js->win32.index = index;
536
537 _glfwInputJoystick(js, GLFW_CONNECTED);
538 }
539 }
540
541 if (_glfw.win32.dinput8.api)
542 {
543 if (FAILED(IDirectInput8_EnumDevices(_glfw.win32.dinput8.api,
544 DI8DEVCLASS_GAMECTRL,
545 deviceCallback,
546 NULL,
547 DIEDFL_ALLDEVICES)))
548 {
549 _glfwInputError(GLFW_PLATFORM_ERROR,
550 "Failed to enumerate DirectInput8 devices");
551 return;
552 }
553 }
554}

556// Checks for joystick disconnection after DBT_DEVICEREMOVECOMPLETE
557//
558void _glfwDetectJoystickDisconnectionWin32(void)
559{
560 int jid;
561
562 for (jid = 0; jid <= GLFW_JOYSTICK_LAST; jid++)
563 {
564 _GLFWjoystick* js = _glfw.joysticks + jid;
565 if (js->connected)
566 _glfwPollJoystickWin32(js, _GLFW_POLL_PRESENCE);
567 }
568}

606GLFWbool _glfwPollJoystickWin32(_GLFWjoystick* js, int mode)
607{
608 if (js->win32.device)
609 {
610 int i, ai = 0, bi = 0, pi = 0;
611 HRESULT result;
612 DIJOYSTATE state = {0};
613
614 IDirectInputDevice8_Poll(js->win32.device);
615 result = IDirectInputDevice8_GetDeviceState(js->win32.device,
616 sizeof(state),
617 &state);
618 if (result == DIERR_NOTACQUIRED || result == DIERR_INPUTLOST)
619 {
620 IDirectInputDevice8_Acquire(js->win32.device);
621 IDirectInputDevice8_Poll(js->win32.device);
622 result = IDirectInputDevice8_GetDeviceState(js->win32.device,
623 sizeof(state),
624 &state);
625 }
626
627 if (FAILED(result))
628 {
629 closeJoystick(js);
630 return GLFW_FALSE;
631 }
632
633 if (mode == _GLFW_POLL_PRESENCE)
634 return GLFW_TRUE;
635
636 for (i = 0; i < js->win32.objectCount; i++)
637 {
638 const void* data = (char*) &state + js->win32.objects[i].offset;
639
640 switch (js->win32.objects[i].type)
641 {
642 case _GLFW_TYPE_AXIS:
643 case _GLFW_TYPE_SLIDER:
644 {
645 const float value = (*((LONG*) data) + 0.5f) / 32767.5f;
646 _glfwInputJoystickAxis(js, ai, value);
647 ai++;
648 break;
649 }
650
651 case _GLFW_TYPE_BUTTON:
652 {
653 const char value = (*((BYTE*) data) & 0x80) != 0;
654 _glfwInputJoystickButton(js, bi, value);
655 bi++;
656 break;
657 }
658
659 case _GLFW_TYPE_POV:
660 {
661 const int states[9] =
662 {
663 GLFW_HAT_UP,
664 GLFW_HAT_RIGHT_UP,
665 GLFW_HAT_RIGHT,
666 GLFW_HAT_RIGHT_DOWN,
667 GLFW_HAT_DOWN,
668 GLFW_HAT_LEFT_DOWN,
669 GLFW_HAT_LEFT,
670 GLFW_HAT_LEFT_UP,
671 GLFW_HAT_CENTERED
672 };
673
674 // Screams of horror are appropriate at this point
675 int stateIndex = LOWORD(*(DWORD*) data) / (45 * DI_DEGREES);
676 if (stateIndex < 0 || stateIndex > 8)
677 stateIndex = 8;
678
679 _glfwInputJoystickHat(js, pi, states[stateIndex]);
680 pi++;
681 break;
682 }
683 }
684 }
685 }
686 else
687 {
688 int i, dpad = 0;
689 DWORD result;
690 XINPUT_STATE xis;
691 const WORD buttons[10] =
692 {
693 XINPUT_GAMEPAD_A,
694 XINPUT_GAMEPAD_B,
695 XINPUT_GAMEPAD_X,
696 XINPUT_GAMEPAD_Y,
697 XINPUT_GAMEPAD_LEFT_SHOULDER,
698 XINPUT_GAMEPAD_RIGHT_SHOULDER,
699 XINPUT_GAMEPAD_BACK,
700 XINPUT_GAMEPAD_START,
701 XINPUT_GAMEPAD_LEFT_THUMB,
702 XINPUT_GAMEPAD_RIGHT_THUMB
703 };
704
705 result = XInputGetState(js->win32.index, &xis);
706 if (result != ERROR_SUCCESS)
707 {
708 if (result == ERROR_DEVICE_NOT_CONNECTED)
709 closeJoystick(js);
710
711 return GLFW_FALSE;
712 }
713
714 if (mode == _GLFW_POLL_PRESENCE)
715 return GLFW_TRUE;
716
717 _glfwInputJoystickAxis(js, 0, (xis.Gamepad.sThumbLX + 0.5f) / 32767.5f);
718 _glfwInputJoystickAxis(js, 1, -(xis.Gamepad.sThumbLY + 0.5f) / 32767.5f);
719 _glfwInputJoystickAxis(js, 2, (xis.Gamepad.sThumbRX + 0.5f) / 32767.5f);
720 _glfwInputJoystickAxis(js, 3, -(xis.Gamepad.sThumbRY + 0.5f) / 32767.5f);
721 _glfwInputJoystickAxis(js, 4, xis.Gamepad.bLeftTrigger / 127.5f - 1.f);
722 _glfwInputJoystickAxis(js, 5, xis.Gamepad.bRightTrigger / 127.5f - 1.f);
723
724 for (i = 0; i < 10; i++)
725 {
726 const char value = (xis.Gamepad.wButtons & buttons[i]) ? 1 : 0;
727 _glfwInputJoystickButton(js, i, value);
728 }
729
730 if (xis.Gamepad.wButtons & XINPUT_GAMEPAD_DPAD_UP)
731 dpad |= GLFW_HAT_UP;
732 if (xis.Gamepad.wButtons & XINPUT_GAMEPAD_DPAD_RIGHT)
733 dpad |= GLFW_HAT_RIGHT;
734 if (xis.Gamepad.wButtons & XINPUT_GAMEPAD_DPAD_DOWN)
735 dpad |= GLFW_HAT_DOWN;
736 if (xis.Gamepad.wButtons & XINPUT_GAMEPAD_DPAD_LEFT)
737 dpad |= GLFW_HAT_LEFT;
738
739 _glfwInputJoystickHat(js, 0, dpad);
740 }
741
742 return GLFW_TRUE;
743}
