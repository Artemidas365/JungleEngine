192 template<typename T, qualifier Q>
193 GLM_FUNC_QUALIFIER qua<T, Q>::qua(vec<3, T, Q> const& u, vec<3, T, Q> const& v)
194 {
195 T norm_u_norm_v = sqrt(dot(u, u) * dot(v, v));
196 T real_part = norm_u_norm_v + dot(u, v);
197 vec<3, T, Q> t;
198
199 if(real_part < static_cast<T>(1.e-6f) * norm_u_norm_v)
200 {
201 // If u and v are exactly opposite, rotate 180 degrees
202 // around an arbitrary orthogonal axis. Axis normalisation
203 // can happen later, when we normalise the quaternion.
204 real_part = static_cast<T>(0);
205 t = abs(u.x) > abs(u.z) ? vec<3, T, Q>(-u.y, u.x, static_cast<T>(0)) : vec<3, T, Q>(static_cast<T>(0), -u.z, u.y);
206 }
207 else
208 {
209 // Otherwise, build quaternion the standard way.
210 t = cross(u, v);
211 }
212
213 *this = normalize(qua<T, Q>::wxyz(real_part, t.x, t.y, t.z));
214 }

216 template<typename T, qualifier Q>
217 GLM_FUNC_QUALIFIER GLM_CONSTEXPR qua<T, Q>::qua(vec<3, T, Q> const& eulerAngle)
218 {
219 vec<3, T, Q> c = glm::cos(eulerAngle * T(0.5));
220 vec<3, T, Q> s = glm::sin(eulerAngle * T(0.5));
221
222 this->w = c.x * c.y * c.z + s.x * s.y * s.z;
223 this->x = s.x * c.y * c.z - c.x * s.y * s.z;
224 this->y = c.x * s.y * c.z + s.x * c.y * s.z;
225 this->z = c.x * c.y * s.z - s.x * s.y * c.z;
226 }

298 qua<T, Q> const q(r);
299
