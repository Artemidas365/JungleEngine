40// Callback for EnumDisplayMonitors in createMonitor
41//
42static BOOL CALLBACK monitorCallback(HMONITOR handle,
43 HDC dc,
44 RECT* rect,
45 LPARAM data)
46{
47 MONITORINFOEXW mi;
48 ZeroMemory(&mi, sizeof(mi));
49 mi.cbSize = sizeof(mi);
50
51 if (GetMonitorInfoW(handle, (MONITORINFO*) &mi))
52 {
53 _GLFWmonitor* monitor = (_GLFWmonitor*) data;
54 if (wcscmp(mi.szDevice, monitor->win32.adapterName) == 0)
55 monitor->win32.handle = handle;
56 }
57
58 return TRUE;
59}

61// Create monitor from an adapter and (optionally) a display
62//
63static _GLFWmonitor* createMonitor(DISPLAY_DEVICEW* adapter,
64 DISPLAY_DEVICEW* display)
65{
66 _GLFWmonitor* monitor;
67 int widthMM, heightMM;
68 char* name;
69 HDC dc;
70 DEVMODEW dm;
71 RECT rect;
72
73 if (display)
74 name = _glfwCreateUTF8FromWideStringWin32(display->DeviceString);
75 else
76 name = _glfwCreateUTF8FromWideStringWin32(adapter->DeviceString);
77 if (!name)
78 return NULL;
79
80 ZeroMemory(&dm, sizeof(dm));
81 dm.dmSize = sizeof(dm);
82 EnumDisplaySettingsW(adapter->DeviceName, ENUM_CURRENT_SETTINGS, &dm);
83
84 dc = CreateDCW(L"DISPLAY", adapter->DeviceName, NULL, NULL);
85
86 if (IsWindows8Point1OrGreater())
87 {
88 widthMM = GetDeviceCaps(dc, HORZSIZE);
89 heightMM = GetDeviceCaps(dc, VERTSIZE);
90 }
91 else
92 {
93 widthMM = (int) (dm.dmPelsWidth * 25.4f / GetDeviceCaps(dc, LOGPIXELSX));
94 heightMM = (int) (dm.dmPelsHeight * 25.4f / GetDeviceCaps(dc, LOGPIXELSY));
95 }
96
97 DeleteDC(dc);
98
99 monitor = _glfwAllocMonitor(name, widthMM, heightMM);
100 _glfw_free(name);
101
102 if (adapter->StateFlags & DISPLAY_DEVICE_MODESPRUNED)
103 monitor->win32.modesPruned = GLFW_TRUE;
104
105 wcscpy(monitor->win32.adapterName, adapter->DeviceName);
106 WideCharToMultiByte(CP_UTF8, 0,
107 adapter->DeviceName, -1,
108 monitor->win32.publicAdapterName,
109 sizeof(monitor->win32.publicAdapterName),
110 NULL, NULL);
111
112 if (display)
113 {
114 wcscpy(monitor->win32.displayName, display->DeviceName);
115 WideCharToMultiByte(CP_UTF8, 0,
116 display->DeviceName, -1,
117 monitor->win32.publicDisplayName,
118 sizeof(monitor->win32.publicDisplayName),
119 NULL, NULL);
120 }
121
122 rect.left = dm.dmPosition.x;
123 rect.top = dm.dmPosition.y;
124 rect.right = dm.dmPosition.x + dm.dmPelsWidth;
125 rect.bottom = dm.dmPosition.y + dm.dmPelsHeight;
126
127 EnumDisplayMonitors(NULL, &rect, monitorCallback, (LPARAM) monitor);
128 return monitor;
129}

136// Poll for changes in the set of connected monitors
137//
138void _glfwPollMonitorsWin32(void)
139{
140 int i, disconnectedCount;
141 _GLFWmonitor** disconnected = NULL;
142 DWORD adapterIndex, displayIndex;
143 DISPLAY_DEVICEW adapter, display;
144 _GLFWmonitor* monitor;
145
146 disconnectedCount = _glfw.monitorCount;
147 if (disconnectedCount)
148 {
149 disconnected = _glfw_calloc(_glfw.monitorCount, sizeof(_GLFWmonitor*));
150 memcpy(disconnected,
151 _glfw.monitors,
152 _glfw.monitorCount * sizeof(_GLFWmonitor*));
153 }
154
155 for (adapterIndex = 0; ; adapterIndex++)
156 {
157 int type = _GLFW_INSERT_LAST;
158
159 ZeroMemory(&adapter, sizeof(adapter));
160 adapter.cb = sizeof(adapter);
161
162 if (!EnumDisplayDevicesW(NULL, adapterIndex, &adapter, 0))
163 break;
164
165 if (!(adapter.StateFlags & DISPLAY_DEVICE_ACTIVE))
166 continue;
167
168 if (adapter.StateFlags & DISPLAY_DEVICE_PRIMARY_DEVICE)
169 type = _GLFW_INSERT_FIRST;
170
171 for (displayIndex = 0; ; displayIndex++)
172 {
173 ZeroMemory(&display, sizeof(display));
174 display.cb = sizeof(display);
175
176 if (!EnumDisplayDevicesW(adapter.DeviceName, displayIndex, &display, 0))
177 break;
178
179 if (!(display.StateFlags & DISPLAY_DEVICE_ACTIVE))
180 continue;
181
182 for (i = 0; i < disconnectedCount; i++)
183 {
184 if (disconnected[i] &&
185 wcscmp(disconnected[i]->win32.displayName,
186 display.DeviceName) == 0)
187 {
188 disconnected[i] = NULL;
189 // handle may have changed, update
190 EnumDisplayMonitors(NULL, NULL, monitorCallback, (LPARAM) _glfw.monitors[i]);
191 break;
192 }
193 }
194
195 if (i < disconnectedCount)
196 continue;
197
198 monitor = createMonitor(&adapter, &display);
199 if (!monitor)
200 {
201 _glfw_free(disconnected);
202 return;
203 }
204
205 _glfwInputMonitor(monitor, GLFW_CONNECTED, type);
206
207 type = _GLFW_INSERT_LAST;
208 }
209
210 // HACK: If an active adapter does not have any display devices
211 // (as sometimes happens), add it directly as a monitor
212 if (displayIndex == 0)
213 {
214 for (i = 0; i < disconnectedCount; i++)
215 {
216 if (disconnected[i] &&
217 wcscmp(disconnected[i]->win32.adapterName,
218 adapter.DeviceName) == 0)
219 {
220 disconnected[i] = NULL;
221 break;
222 }
223 }
224
225 if (i < disconnectedCount)
226 continue;
227
228 monitor = createMonitor(&adapter, NULL);
229 if (!monitor)
230 {
231 _glfw_free(disconnected);
232 return;
233 }
234
235 _glfwInputMonitor(monitor, GLFW_CONNECTED, type);
236 }
237 }
238
239 for (i = 0; i < disconnectedCount; i++)
240 {
241 if (disconnected[i])
242 _glfwInputMonitor(disconnected[i], GLFW_DISCONNECTED, 0);
243 }
244
245 _glfw_free(disconnected);
246}

248// Change the current video mode
249//
250void _glfwSetVideoModeWin32(_GLFWmonitor* monitor, const GLFWvidmode* desired)
251{
252 GLFWvidmode current;
253 const GLFWvidmode* best;
254 DEVMODEW dm;
255 LONG result;
256
257 best = _glfwChooseVideoMode(monitor, desired);
258 _glfwGetVideoModeWin32(monitor, &current);
259 if (_glfwCompareVideoModes(&current, best) == 0)
260 return;
261
262 ZeroMemory(&dm, sizeof(dm));
263 dm.dmSize = sizeof(dm);
264 dm.dmFields = DM_PELSWIDTH | DM_PELSHEIGHT | DM_BITSPERPEL |
265 DM_DISPLAYFREQUENCY;
266 dm.dmPelsWidth = best->width;
267 dm.dmPelsHeight = best->height;
268 dm.dmBitsPerPel = best->redBits + best->greenBits + best->blueBits;
269 dm.dmDisplayFrequency = best->refreshRate;
270
271 if (dm.dmBitsPerPel < 15 || dm.dmBitsPerPel >= 24)
272 dm.dmBitsPerPel = 32;
273
274 result = ChangeDisplaySettingsExW(monitor->win32.adapterName,
275 &dm,
276 NULL,
277 CDS_FULLSCREEN,
278 NULL);
279 if (result == DISP_CHANGE_SUCCESSFUL)
280 monitor->win32.modeChanged = GLFW_TRUE;
281 else
282 {
283 const char* description = "Unknown error";
284
285 if (result == DISP_CHANGE_BADDUALVIEW)
286 description = "The system uses DualView";
287 else if (result == DISP_CHANGE_BADFLAGS)
288 description = "Invalid flags";
289 else if (result == DISP_CHANGE_BADMODE)
290 description = "Graphics mode not supported";
291 else if (result == DISP_CHANGE_BADPARAM)
292 description = "Invalid parameter";
293 else if (result == DISP_CHANGE_FAILED)
294 description = "Graphics mode failed";
295 else if (result == DISP_CHANGE_NOTUPDATED)
296 description = "Failed to write to registry";
297 else if (result == DISP_CHANGE_RESTART)
298 description = "Computer restart required";
299
300 _glfwInputError(GLFW_PLATFORM_ERROR,
301 "Win32: Failed to set video mode: %s",
302 description);
303 }
304}

318void _glfwGetHMONITORContentScaleWin32(HMONITOR handle, float* xscale, float* yscale)
319{
320 UINT xdpi, ydpi;
321
322 if (xscale)
323 *xscale = 0.f;
324 if (yscale)
325 *yscale = 0.f;
326
327 if (IsWindows8Point1OrGreater())
328 {
329 if (GetDpiForMonitor(handle, MDT_EFFECTIVE_DPI, &xdpi, &ydpi) != S_OK)
330 {
331 _glfwInputError(GLFW_PLATFORM_ERROR, "Win32: Failed to query monitor DPI");
332 return;
333 }
334 }
335 else
336 {
337 const HDC dc = GetDC(NULL);
338 xdpi = GetDeviceCaps(dc, LOGPIXELSX);
339 ydpi = GetDeviceCaps(dc, LOGPIXELSY);
340 ReleaseDC(NULL, dc);
341 }
342
343 if (xscale)
344 *xscale = xdpi / (float) USER_DEFAULT_SCREEN_DPI;
345 if (yscale)
346 *yscale = ydpi / (float) USER_DEFAULT_SCREEN_DPI;
347}

398GLFWvidmode* _glfwGetVideoModesWin32(_GLFWmonitor* monitor, int* count)
399{
400 int modeIndex = 0, size = 0;
401 GLFWvidmode* result = NULL;
402
403 *count = 0;
404
405 for (;;)
406 {
407 int i;
408 GLFWvidmode mode;
409 DEVMODEW dm;
410
411 ZeroMemory(&dm, sizeof(dm));
412 dm.dmSize = sizeof(dm);
413
414 if (!EnumDisplaySettingsW(monitor->win32.adapterName, modeIndex, &dm))
415 break;
416
417 modeIndex++;
418
419 // Skip modes with less than 15 BPP
420 if (dm.dmBitsPerPel < 15)
421 continue;
422
423 mode.width = dm.dmPelsWidth;
424 mode.height = dm.dmPelsHeight;
425 mode.refreshRate = dm.dmDisplayFrequency;
426 _glfwSplitBPP(dm.dmBitsPerPel,
427 &mode.redBits,
428 &mode.greenBits,
429 &mode.blueBits);
430
431 for (i = 0; i < *count; i++)
432 {
433 if (_glfwCompareVideoModes(result + i, &mode) == 0)
434 break;
435 }
436
437 // Skip duplicate modes
438 if (i < *count)
439 continue;
440
441 if (monitor->win32.modesPruned)
442 {
443 // Skip modes not supported by the connected displays
444 if (ChangeDisplaySettingsExW(monitor->win32.adapterName,
445 &dm,
446 NULL,
447 CDS_TEST,
448 NULL) != DISP_CHANGE_SUCCESSFUL)
449 {
450 continue;
451 }
452 }
453
454 if (*count == size)
455 {
456 size += 128;
457 result = (GLFWvidmode*) _glfw_realloc(result, size * sizeof(GLFWvidmode));
458 }
459
460 (*count)++;
461 result[*count - 1] = mode;
462 }
463
464 if (!*count)
465 {
466 // HACK: Report the current mode if no valid modes were found
467 result = _glfw_calloc(1, sizeof(GLFWvidmode));
468 _glfwGetVideoModeWin32(monitor, result);
469 *count = 1;
470 }
471
472 return result;
473}

492GLFWbool _glfwGetGammaRampWin32(_GLFWmonitor* monitor, GLFWgammaramp* ramp)
493{
494 HDC dc;
495 WORD values[3][256];
496
497 dc = CreateDCW(L"DISPLAY", monitor->win32.adapterName, NULL, NULL);
498 GetDeviceGammaRamp(dc, values);
499 DeleteDC(dc);
500
501 _glfwAllocGammaArrays(ramp, 256);
502
503 memcpy(ramp->red, values[0], sizeof(values[0]));
504 memcpy(ramp->green, values[1], sizeof(values[1]));
505 memcpy(ramp->blue, values[2], sizeof(values[2]));
506
507 return GLFW_TRUE;
508}

510void _glfwSetGammaRampWin32(_GLFWmonitor* monitor, const GLFWgammaramp* ramp)
511{
512 HDC dc;
513 WORD values[3][256];
514
515 if (ramp->size != 256)
516 {
517 _glfwInputError(GLFW_PLATFORM_ERROR,
518 "Win32: Gamma ramp size must be 256");
519 return;
520 }
521
522 memcpy(values[0], ramp->red, sizeof(values[0]));
523 memcpy(values[1], ramp->green, sizeof(values[1]));
524 memcpy(values[2], ramp->blue, sizeof(values[2]));
525
526 dc = CreateDCW(L"DISPLAY", monitor->win32.adapterName, NULL, NULL);
527 SetDeviceGammaRamp(dc, values);
528 DeleteDC(dc);
529}
