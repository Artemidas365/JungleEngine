54 void EngineCore::processInput() {
55
56 if (glfwGetKey(window, GLFW_KEY_ESCAPE) == GLFW_PRESS)
57 glfwSetWindowShouldClose(window, true);
58
59 if (glfwGetKey(window, GLFW_KEY_F11) == GLFW_PRESS) {
60
61 if (glfwGetWindowMonitor(window) == NULL) {
62 // fullscreen
63 GLFWmonitor *primaryMonitor = glfwGetPrimaryMonitor();
64 const GLFWvidmode *mode = glfwGetVideoMode(primaryMonitor);
65 glfwSetWindowMonitor(window, primaryMonitor, 0, 0, mode->width, mode->height, mode->refreshRate);
66 windowWidth = mode->width;
67 windowHeight = mode->height;
68 } else {
69 // windowed
70 windowWidth = 800;
71 windowHeight = 600;
72 glfwSetWindowMonitor(window, NULL, 100, 100, windowWidth, windowHeight, GLFW_DONT_CARE);
73 }
74 }
75
76 if (glfwGetKey(window, GLFW_KEY_W) == GLFW_PRESS)
77 camera.ProcessKeyboard(FORWARD, deltaTime);
78 if (glfwGetKey(window, GLFW_KEY_S) == GLFW_PRESS)
79 camera.ProcessKeyboard(BACKWARD, deltaTime);
80 if (glfwGetKey(window, GLFW_KEY_A) == GLFW_PRESS)
81 camera.ProcessKeyboard(LEFT, deltaTime);
82 if (glfwGetKey(window, GLFW_KEY_D) == GLFW_PRESS)
83 camera.ProcessKeyboard(RIGHT, deltaTime);
84
85 if(glfwGetMouseButton(window, GLFW_MOUSE_BUTTON_RIGHT) == GLFW_PRESS){
86 double xpos;
87 double ypos;
88 glfwGetCursorPos(window, &xpos, &ypos);
89 glfwSetInputMode(window, GLFW_CURSOR, GLFW_CURSOR_DISABLED);
90
91 if (firstMouse)
92 {
93 lastX = xpos;
94 lastY = ypos;
95 firstMouse = false;
96 }
97
98 float xoffset = xpos - lastX;
99 float yoffset = lastY - ypos; // reversed since y-coordinates go from bottom to top
100
101 lastX = xpos;
102 lastY = ypos;
103
104 camera.ProcessMouseMovement(xoffset, yoffset);
105 }
106 if(glfwGetMouseButton(window, GLFW_MOUSE_BUTTON_RIGHT) == GLFW_RELEASE) {
107 glfwSetInputMode(window, GLFW_CURSOR, GLFW_CURSOR_NORMAL);
108 firstMouse = true;
109 }
110 }

269 GLuint EngineCore::generateTex(const char *fileName) {
270 GLuint texture;
271 glGenTextures(1, &texture);
272 glBindTexture(GL_TEXTURE_2D, texture);
273 // set the texture wrapping/filtering options (on the currently bound texture object)
274 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_REPEAT);
275 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_REPEAT);
276 // set texture filtering parameters
277 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
278 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
279 // load and generate the texture
280 int width, height, nrChannels;
281 stbi_set_flip_vertically_on_load(true);
282 unsigned char *data = stbi_load(concatenate(textureDir, fileName).c_str() , &width, &height, &nrChannels, 0);
283 if (data) {
284 if(nrChannels == 3){
285 glTexImage2D(GL_TEXTURE_2D, 0, GL_RGB, width, height, 0, GL_RGB, GL_UNSIGNED_BYTE, data);
286 glGenerateMipmap(GL_TEXTURE_2D);
287 }
288 else if(nrChannels == 4){
289 glTexImage2D(GL_TEXTURE_2D, 0, GL_RGB, width, height, 0, GL_RGBA, GL_UNSIGNED_BYTE, data);
290 glGenerateMipmap(GL_TEXTURE_2D);
291 }
292 else {
293 std::cout << "image has unknown format with " << nrChannels << " channels." << std::endl;
294 }
295
296 } else {
297 std::cout << "Failed to load texture" << std::endl;
298 }
299 stbi_image_free(data);
300 return texture;
301 }
