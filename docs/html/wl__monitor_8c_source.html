64static void outputHandleMode(void* userData,
65 struct wl_output* output,
66 uint32_t flags,
67 int32_t width,
68 int32_t height,
69 int32_t refresh)
70{
71 struct _GLFWmonitor* monitor = userData;
72 GLFWvidmode mode;
73
74 mode.width = width;
75 mode.height = height;
76 mode.redBits = 8;
77 mode.greenBits = 8;
78 mode.blueBits = 8;
79 mode.refreshRate = (int) round(refresh / 1000.0);
80
81 monitor->modeCount++;
82 monitor->modes =
83 _glfw_realloc(monitor->modes, monitor->modeCount * sizeof(GLFWvidmode));
84 monitor->modes[monitor->modeCount - 1] = mode;
85
86 if (flags & WL_OUTPUT_MODE_CURRENT)
87 monitor->wl.currentMode = monitor->modeCount - 1;
88}

90static void outputHandleDone(void* userData, struct wl_output* output)
91{
92 struct _GLFWmonitor* monitor = userData;
93
94 if (monitor->widthMM <= 0 || monitor->heightMM <= 0)
95 {
96 // If Wayland does not provide a physical size, assume the default 96 DPI
97 const GLFWvidmode* mode = &monitor->modes[monitor->wl.currentMode];
98 monitor->widthMM = (int) (mode->width * 25.4f / 96.f);
99 monitor->heightMM = (int) (mode->height * 25.4f / 96.f);
100 }
101
102 for (int i = 0; i < _glfw.monitorCount; i++)
103 {
104 if (_glfw.monitors[i] == monitor)
105 return;
106 }
107
108 _glfwInputMonitor(monitor, GLFW_CONNECTED, _GLFW_INSERT_LAST);
109}

111static void outputHandleScale(void* userData,
112 struct wl_output* output,
113 int32_t factor)
114{
115 struct _GLFWmonitor* monitor = userData;
116
117 monitor->wl.contentScale = factor;
118
119 for (_GLFWwindow* window = _glfw.windowListHead; window; window = window->next)
120 {
121 for (int i = 0; i < window->wl.scaleCount; i++)
122 {
123 if (window->wl.scales[i].output == monitor->wl.output)
124 {
125 window->wl.scales[i].factor = monitor->wl.contentScale;
126 _glfwUpdateContentScaleWayland(window);
127 break;
128 }
129 }
130 }
131}

167void _glfwAddOutputWayland(uint32_t name, uint32_t version)
168{
169 if (version < 2)
170 {
171 _glfwInputError(GLFW_PLATFORM_ERROR,
172 "Wayland: Unsupported output interface version");
173 return;
174 }
175
176#ifdef WL_OUTPUT_NAME_SINCE_VERSION
177 version = _glfw_min(version, WL_OUTPUT_NAME_SINCE_VERSION);
178#else
179 version = 2;
180#endif
181
182 struct wl_output* output = wl_registry_bind(_glfw.wl.registry,
183 name,
184 &wl_output_interface,
185 version);
186 if (!output)
187 return;
188
189 // The actual name of this output will be set in the geometry handler
190 _GLFWmonitor* monitor = _glfwAllocMonitor("", 0, 0);
191 monitor->wl.contentScale = 1;
192 monitor->wl.output = output;
193 monitor->wl.name = name;
194
195 wl_proxy_set_tag((struct wl_proxy*) output, &_glfw.wl.tag);
196 wl_output_add_listener(output, &outputListener, monitor);
197}
